TYPE "Sm_COMMANDS"
VERSION : 0.1
   STRUCT
      Start : Bool;
      Pause : Bool;
      Stop : Bool;
      Reset : Bool;
      Abort : Bool;
   END_STRUCT;

END_TYPE

TYPE "Sm_FLAGS"
VERSION : 0.1
   STRUCT
      Release : Bool;
      Restart : Bool;
      Degraded : Bool;
   END_STRUCT;

END_TYPE

TYPE "Sm_STATUS"
VERSION : 0.1
   STRUCT
      Busy : Bool;
      StepName : String;
      StatusName : String;
      StepNb : Int;
      SubStepNb : Int;
      LastStepNb : Int;
      LastSubStepNb : Int;
      Status : Int;
   END_STRUCT;

END_TYPE

      Hmi : "Hmi_SM";
   END_STRUCT;

END_TYPE
